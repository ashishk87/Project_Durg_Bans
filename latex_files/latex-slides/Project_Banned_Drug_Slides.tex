\begin{frame}{Background}{What are FDCs?}
	\begin{itemize}
		\item Fixed Dose Combination drugs (FDCs) are combination of two or more drugs in a single pill
		\bigskip
		\pause
		\item The reason to justify two or more drugs in a FDC could stem from various reasons such as:
		\begin{enumerate}[(i)]
			\pause
			\item improvement in therapeutic efficiency
			\pause
			\item diminish the incidence of adverse effects
			of a drug
			\pause
			\item increase drug compliance by reducing pill burden
			\pause
			\item superior in terms of pharmacokinetic attribute
			\pause
			\item shrinking the possibility of drug resistance
			\pause
			\item cost-effectiveness
		\end{enumerate}
	\end{itemize}
\end{frame}


\begin{frame}{Background}{FDCs in India}
	\begin{itemize}
		\pause
		\item In 2016, FDC drug sales were 45-50\% of the overall 1 lakh crore sales recorded in the pharmaceutical sector
		\medskip
		\pause
		\item \alert{Srinivasan (2018)} estimate that in 2016 close to 50\% of the FDCs were irrational drugs amounting to 22k-25k crore of overall sales
%		\medskip
%		\item No scientific evidence was provided to CDSCO to prove their therapeutic feasibility
	\end{itemize}
\end{frame}

\begin{frame}{Background}{What factors led to proliferation of FDCs?}
	\begin{enumerate}
		\item Demand side factors
		\bigskip
		\item Supply side factors
		\bigskip
		\item Regulatory environment
	\end{enumerate}
\end{frame}

\begin{frame}{Demand side factors}
	\begin{itemize}
		\item Certain disease conditions warrants the use of combination of various single dose drug formulations in fixed ratios, to defeat the causative drug underlying that disease
		\medskip
		\pause
		\item Example: Cardiovascular diseases, diabetes, and other chronic drugs are best served through combination of multiple drug formulations
		\medskip
		\pause
		\item \alert{Other factors}: costs, compliance and adherence
	\end{itemize}
\end{frame}

\begin{frame}{Supply side factors}
	\begin{itemize}
		\item FDCs are cost effective in terms of production, packaging, and distribution related aspects
		\pause
		\medskip
		\item TRIPS agreement in 2005 + Price controls \pause  $\boldsymbol{\Rightarrow}$ dent in the profit stream of pharma companies 
		\medskip
		\pause
		\item For sellers, combining certain drugs in fixed proportions and relabeling them as a new product is way cheaper than spending billions of rupees in research to invent a new drug 
		\pause
		\medskip
		\item Circumventing the price controls
	\end{itemize}
\end{frame}

\begin{frame}{Regulatory Lapses}
	\begin{itemize}
		\item Disconnect between state and central drug regulatory agencies
		\medskip
		\pause
		\item FDCs were present in Indian market from 1960 onward, but regulators took cognizance of their presence in 1988 when they sought to first regulate them by making changes to the Drug \& Cosmetic Act, 1940
		\medskip
		\pause
		\item As per the Rule 122E of Drugs and Cosmetics Act, FDCs are considered as New Drugs and CDSCO after careful examination grants or denies approval
	\end{itemize}
\end{frame}


\begin{frame}{Regulatory Lapses}
	\begin{itemize}
		\item Only after CDSCO’s approval, a State Licensing Authority (SLA) can issue license for the manufacturing, import, marketing, or sales of the concerned drug
		\medskip
		\pause
		\item However, SLA’s all over the country granted approvals to FDCs on their own
		\medskip
		\pause
		\item SLA’s themselves lacked any technical expertise or access to resources to examine the validity of the drugs, and they did not ask for CDSCO issued NOC from companies
		\medskip
		\pause
		\item That coordination-failure and disconnect between SLA’s and CDSCO was extensively used by companies to introduce many FDCs into market that even lacked therapeutic justification

	\end{itemize}
\end{frame}

\begin{frame}{Government Action}
	\begin{itemize}
		\item Government scrutiny and pressure from NGOs $\boldsymbol{\Rightarrow}$ formation of Kokate Committee in 2014
		\medskip
		\pause
		\item Kokate committee submitted its report in February, 2016 and classified FDC drugs into three categories. Out of 6,220 FDCs
		\pause
		\begin{enumerate}[(i)]
			\item 15 percent were outrightly labeled as irrational drugs representing 344 FDCs
			\pause
			\item 30 percent either required more data or further careful examination
			\pause
			\item Rest 42\% of the drugs were deemed to be rational based on the available scientific evidence and literature
		\end{enumerate}
			\medskip
	\pause
	\item Prompted by the committee’s recommendation, government banned the prescribed irrational FDC
	drugs in March, 2016
	\end{itemize}
\end{frame}

\begin{frame}{Main focus of our study}
	\alert{Research Question:} How did firms in pharmaceutical sector respond to this external shock where GOI suddenly banned the marketing, sales and distribution of 344 FDCs in March-2016.
	\pause
	\medskip
	\begin{itemize}
		\item Our study adds to the literature on the process of technological abandonment prompted by the efficacy concerns
		\pause
		\medskip
		\item On what factors did firm base their exit decisions?
		\pause
		\medskip
		\item Does the abandonment process vary based on product market attributes?
	\end{itemize}
\end{frame}

	\begin{frame}{Empirical Framework}{Identification Strategy}
	\begin{itemize}
		\pause
		\item Treating the March-2016 ban as an exogenous event (or at least its timing)
		\pause
		\medskip
		\item Examine the firm-level response to the En-Masse ban
		\medskip
		\pause
		\item We employ triple difference-in-difference estimator and event study design framework for our analysis
	\end{itemize}
\end{frame}

\begin{frame}{Treatment v/s Control group}
	\begin{itemize}
		\item \alert{Treatment group:} The AIOCD data set have information on 96 FDCs  out of 344 FDCs that were banned. These 96 molecules form our treatment group.
		\medskip
		\pause
		\item \alert{Control group:} All other molecules which falls under the same therapeutic category, have same mode of action on targets, and share same mechanism of action
		based on route of administration.
	\end{itemize}
\end{frame}


\section{Main Findings}

\begin{frame}{Reputation Costs}
	\begin{itemize}
		\item When faced with a negative information shock firms face a trade off in deciding between ethics and profits
		\pause
		\medskip
		\item Selling banned drugs entail a reputation costs especially for MNCs
		\pause
		\medskip
		\item We should expect players with more reputation costs showing a higher degree of responsiveness to the ban
	\end{itemize}
\end{frame}

\section{Heterogeneity in firm response in molecule markets with varying elasticity}
